```java
public class HttpTransport {

    private final HttpClient client;
    private final String endpoint;

    public HttpTransport(String endpoint, String apiKey, long timeout) {
        this.endpoint = endpoint;
        this.client = HttpClient.newBuilder()
                .connectTimeout(Duration.ofMillis(timeout))
                .build();
    }

    public <T> T post(String path, Object req, Class<T> clazz) {
        try {
            HttpRequest request = HttpRequest.newBuilder()
                    .uri(URI.create(endpoint + path))
                    .POST(HttpRequest.BodyPublishers.ofString(Json.encode(req)))
                    .header("Content-Type", "application/json")
                    .build();

            HttpResponse<String> response =
                    client.send(request, HttpResponse.BodyHandlers.ofString());

            return Json.decode(response.body(), clazz);

        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }

    public GenerationStream stream(String path, Object req) {
        return new HttpStreamClient(client, endpoint + path, req);
    }

    public <T> T get(String path, Class<T> clazz) {
        try {
            HttpRequest request = HttpRequest.newBuilder()
                    .uri(URI.create(endpoint + path))
                    .GET()
                    .build();

            HttpResponse<String> response =
                    client.send(request, HttpResponse.BodyHandlers.ofString());

            return Json.decode(response.body(), clazz);

        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }
}
```

```java
public class HttpStreamClient implements GenerationStream {

    private final List<TokenListener> tokenListeners = new CopyOnWriteArrayList<>();
    private final List<Runnable> completeListeners = new CopyOnWriteArrayList<>();
    private final List<ErrorListener> errorListeners = new CopyOnWriteArrayList<>();

    public HttpStreamClient(HttpClient client, String url, Object request) {

        CompletableFuture.runAsync(() -> {
            try {
                HttpRequest httpRequest = HttpRequest.newBuilder()
                        .uri(URI.create(url))
                        .POST(HttpRequest.BodyPublishers.ofString(Json.encode(request)))
                        .build();

                client.send(httpRequest, HttpResponse.BodyHandlers.ofLines())
                        .body()
                        .forEach(line -> {
                            if (line.startsWith("data:")) {
                                String token = line.substring(5);
                                for (var l : tokenListeners) l.onToken(token);
                            }
                        });

                completeListeners.forEach(Runnable::run);

            } catch (Exception e) {
                for (var l : errorListeners) l.onError(e);
            }
        });
    }

    @Override
    public void onToken(TokenListener listener) {
        tokenListeners.add(listener);
    }

    @Override
    public void onComplete(Runnable listener) {
        completeListeners.add(listener);
    }

    @Override
    public void onError(ErrorListener listener) {
        errorListeners.add(listener);
    }

    @Override
    public void cancel() {
        // implement cancellation token if needed
    }
}
```
